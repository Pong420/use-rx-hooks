# useRxInput

```js
const [value, inputProps] = useRxInput(options?);
```

## Typings

```js

type TargetEl = HTMLInputElement | HTMLTextAreaElement;

interface InputProps<T extends TargetEl = HTMLInputElement> {
  ref: RefObject<T>;
  defaultValue?: string;
}

type RxInputPipe<O> = (ob: Observable<string | undefined>) => Observable<O>;

interface UseRxInputOptions<O, T extends TargetEl = HTMLInputElement> {
  interceptors?: (el: T) => (ob: Observable<Event>) => Observable<Event>;
  defaultValue?: string;
  pipe?: RxInputPipe<O>;
}

declare function useRxInput<O, T extends TargetEl = HTMLInputElement>(
  options?:UseRxInputOptions<O>
): [O, InputProps];
```
