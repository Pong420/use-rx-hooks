# useRxFileToImage

```js
const [state?, props] = useRxFileToImage<T extends HTMLElement, E extends Event>(fn: RxFileToImageInput<E>);

// HOC for `useRxFileToImage`

```

## Typings

```js
import { FromEventTarget } from 'rxjs/internal/observable/fromEvent';

export type RxFileToImageInput<E> = (
  target: FromEventTarget<E>

interface State {
  file: File;
  url: string;
```
